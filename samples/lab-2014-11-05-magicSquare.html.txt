         function btnAddClick() {

         function btnMultClick() {
            doTheMath("mult");
         }

         function btnSubClick() {
            doTheMath("sub");
         }

         function doTheMath(operation) {
            /*
               The GUI is structured in a 3 by 3 grid, numbered as follows:
                   1  2  3
                   4  5  6
                   7  8  9
             */

            // Get the values the users entered.
            var valueAtPos1 = document.getElementById("txt1").value;
            valueAtPos1 = parseInt(valueAtPos1);                                    
            var valueAtPos2 = parseInt(document.getElementById("txt2").value);
            var valueAtPos4 = parseInt(document.getElementById("txt4").value);
            var valueAtPos5 = parseInt(document.getElementById("txt5").value);

            // Do the math.
            var valueForPos3 = NaN;  // Initialize to NaN (Not a Number) 
            var valueForPos6 = NaN;  // to insure that we are not accidentally 
            var valueForPos7 = NaN;  // setting values that look like realistic
            var valueForPos8 = NaN;  // results of the calculations.
            
            if (operation === "add") {
              valueForPos3 = valueAtPos1 + valueAtPos2;
              valueForPos6 = valueAtPos4 + valueAtPos5;
              valueForPos7 = valueAtPos1 + valueAtPos4;
              valueForPos8 = valueAtPos2 + valueAtPos5;
            } else if (operation === "mult") {
              valueForPos3 = valueAtPos1 * valueAtPos2;
              valueForPos6 = valueAtPos4 * valueAtPos5;
              valueForPos7 = valueAtPos1 * valueAtPos4;
              valueForPos8 = valueAtPos2 * valueAtPos5;
            } else if (operation === "sub") {
              valueForPos3 = valueAtPos1 - valueAtPos2;
              valueForPos6 = valueAtPos4 - valueAtPos5;
              valueForPos7 = valueAtPos1 - valueAtPos4;
              valueForPos8 = valueAtPos2 - valueAtPos5;
            }
            
            // Show the results.
            document.getElementById("txt3").value = valueForPos3;
            document.getElementById("txt6").value = valueForPos6;
            document.getElementById("txt7").value = valueForPos7;
            document.getElementById("txt8").value = valueForPos8;
         }
